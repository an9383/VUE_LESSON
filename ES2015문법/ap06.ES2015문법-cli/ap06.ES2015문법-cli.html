################################## # node / npm 버전 확인 및 업그레이드 ################################## $ node -v # node 버전 확인 v16.16.0 $ npm -v # npm 버전 확인 9.1.1 $ npx -v # npx(=npm extension) 버전 확인 9.1.1 $ npm install -g npm@latest # npm 최신 버전 설치. -g: 전역으로 설치하라는 의미. ################################## # node 프로젝트 폴더 만들기 ################################## $ ls # 현재 작업 디렉토리 파일 목록 조회 ES2015babel방식.html src/ $ pwd # 현재 작업 디렉토리 위치 확인 /c/LectureMaterial/ES2015문법/ap06.ES2015문법-cli # 현재 디렉토리에 es2015test 폴더 만들기 $ mkdir -p es2015test # 현재 작업 디렉토리 파일 목록 조회 $ ls ap06.ES2015문법-cli.html es2015test/ src/ # 만들어진 es2015test 디렉토리로 이동 $ cd es2015test # 현재 작업 디렉토리 확인 $ pwd /c/LectureMaterial/ES2015문법/ap06.ES2015문법-cli/es2015test ################################## # node 프로젝트 설정 초기화 ################################## # 현재 작업 디렉토리 파일 목록 조회 $ ls # node 프로젝트 생성. $ npm init # package.json 파일 생성 됨. nodemon 모듈 설치 ################################## nodemon은 디렉토리의 파일 변경이 감지되면 노드 응용 프로그램을 자동으로 재시작하는 모듈이다. https://www.npmjs.com/package/nodemon 설치하면 node_modules 폴더에 다운로드됨. $ npm install --save-dev nodemon ################################## # node 프로젝트에 babel 라이브러리 설치 # babel 은 자바스크립트 버전 컴파일러다. # ES2015 ===> ES5, JSX ===> ES5 ################################## https://babeljs.io/docs/en/babel-preset-env-standalone https://babeljs.io/docs/en/babel-preset-react https://www.daleseo.com/js-babel/ 설치하면 node_modules 폴더에 다운로드됨. $ npm install --save-dev @babel/core $ npm install --save-dev @babel/cli $ npm install --save-dev @babel/preset-env $ cat package.json { "name": "es2015test", "version": "1.0.0", "description": "", "main": "index.js", "scripts": { "test": "echo \"Error: no test specified\" && exit 1" }, "author": "", "license": "ISC", "devDependencies": { "nodemon": "2.0.20", "@babel/cli": "7.18.6", "@babel/core": "7.18.6", "@babel/preset-env": "7.18.6" } } $ ls -a # 현재 디렉토리의 파일 목록 조회 ./ ../ node_modules/ package.json package-lock.json ################################## # 폴더 구조 및 파일 확인 ################################## \es2015test | package.json |---node_modules/ `---src/ `---ap06-01.js ################################## # node 프로젝트에 src 작업 디렉토리 만들기 ################################## # -p 는 부모 폴더가 없으면 부모 폴더도 만들라는 옵션 $ mkdir -p ./src/ # 상위 디렉토리의 src 디렉토리를 현재 디렉토리로 복사하기. $ cp -r ../src ./ 또는 직접 상위 디렉토리의 src 디렉토리를 현재 디렉토리로 복사하거나 이동한다. $ ls -a ./ ../ node_modules/ package.json package-lock.json src/ $ es2015test/src/ap06-01.js 파일 생성 후 코드 추가 'use strict'; const name = 'world'; console.log(`Hello ${name}`); const getMessage = (params) => { console.log(`Hello ${params}`); }; getMessage('sejoong'); const str = "aaaa"; console.log(str); ################################## # ap06-01.js 디버깅 하기 # 디버깅시 자주 사용되는 단축키 # F5: # F8: # F10: # F11: ################################## ap06-01.js 파일 안에서 F5 버튼을 디버깅을 실행한다. ################################## # 바벨을 이용하여 (트랜스) 컴파일 실행 보기 # ==> 명령어 옵션(--presets=@babel/env) 을 이용한 트랜스 컴파일 수행 ################################## $ npx babel --presets=@babel/env src/ap06-01.js 'use strict'; var name = 'world'; console.log("Hello ".concat(name)); var getMessage = function getMessage(params) { console.log("Hello ".concat(params)); }; getMessage('sejoong'); ################################## # Babel 설정 파일(.babelrc)을 이용한 트랜스 컴파일 수행 ################################## # 폴더 구조 및 파일 확인 \es2015test | .babelrc | package.json |---node_modules/ `---src/ `---ap06-01.js $ touch .babelrc # babel 설정 파일, .babelrc 생성 $ ls -a # 현재 디렉토리의 파일 목록 조회 ./ ../ .babelrc node_modules/ package.json package-lock.json src/ $ echo -e "{\n \"presets\": [\"@babel/env\"]\n}" > .babelrc $ cat .babelrc # .babelrc 내용 확인 { "presets": ["@babel/env"] } # 이전에 실행 컴파일 명령어: npx babel --presets=@babel/env src/ap06-01.js $ npx babel src/ap06-01.js 'use strict'; var name = 'world'; console.log("Hello ".concat(name)); var getMessage = function getMessage(params) { console.log("Hello ".concat(params)); }; getMessage('sejoong') ################################## # src 폴더에 있는 js 파일을 es5로 컴파일하여 # 컴파일된 결과물이 build 폴더에 파일로 추가되도록 하시오. ################################## # package.json 파일이 있는지 확인. $ ls -a ./ ../ .babelrc node_modules/ package.json package-lock.json src/ # packages.json 의 scripts 에 추가된 build 실행하기 $ npm run build ################################## # ==> src 디렉토리의 소스 변경을 자동으로 감지하여 ES5 컴파일하는 설정. # packages.json 의 scripts 에 start 추가 # "start": "nodemon --watch src/ --exec npm run build" ################################## $ ls -a #
