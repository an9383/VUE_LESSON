'warn' : 'off', 'no-debugger': process.env.NODE_ENV === 'production' ? 'warn' : 'off', 'prettier/prettier': ['error', { endOfLine: 'auto' }], 'linebreak-style': 'off', 'arrow-parens': ['error', 'always'], 'arrow-body-style': 'off', 'consistent-return': 'off', 'func-names': 'off', 'no-shadow': 'off', 'no-alert': 'off', 'no-var': 'error', 'no-unused-vars': 'off', 'no-param-reassign': ['error', { props: false }], }, overrides: [ { files: ['**/__tests__/*.{j,t}s?(x)', '**/tests/unit/**/*.spec.{j,t}s?(x)'], env: { jest: true, }, }, ], }; step 4-02.
